### Create legislators table

```sql
CREATE TABLE "public"."legislators" (
  "id" serial,
  "last_name" text NOT NULL,
  "first_name" text NOT NULL,
  "middle_name" text,
  "suffix" text,
  "nickname" text,
  "full_name" text NOT NULL,
  "birthday" date NOT NULL,
  "gender" text,
  "type" varchar(3) NOT NULL,
  "state" varchar(2) NOT NULL,
  "district" integer,
  "party" text,
  "url" text,
  "address" text,
  "phone" text,
  "contact_form" text,
  "twitter" text,
  "facebook" text,
  "youtube" text,
  "youtube_id" text,
  "bioguide_id" text,
  "thomas_id" text,
  "opensecrets_id" text,
  "lis_id" text,
  "cspan_id" text,
  "govtrack_id" text,
  "votesmart_id" text,
  "ballotpedia_id" text,
  "wikipedia_id" text,
  PRIMARY KEY ("id")
);
```
